#include <iostream>
#include <unordered_map>
#include <string>

using namespace std;

void alertLowStock(const unordered_map<string, int>& stockLevels, int threshold) {
    for (const auto& item : stockLevels) {
        if (item.second < threshold) {
            cout << "ALERT: Replenish stock for '" << item.first 
                 << "' - Remaining: " << item.second << endl;
        }
    }
}

int main() {
    unordered_map<string, int> stock = {
        {"iPhone 13", 20},
        {"Redmi Note 12", 4},
        {"Samsung M14", 0}
    };
    int threshold = 5;
    alertLowStock(stock, threshold);
    return 0;
}
